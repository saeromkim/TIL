#### count를 사용해 찾는 방법
```c
#include <iostream>
#include <set>
using namespace std;
int main() {
    int n;
    scanf("%d",&n);

    set<int> s;
    for (int i=0; i<n; i++) {
        int x;
        scanf("%d",&x);
        s.insert(x);
    }

    int m;
    scanf("%d",&m);

    for (int i=0; i<m; i++) {
        int x;
        scanf("%d",&x);
        printf("%d ",s.count(x));
    }

    printf("\n");

    return 0;
}
```
#### 이진탐색 알고리즘 이용해서 푸는 방법
- binary_search 즉 이진탐색이란 정렬돼 있는 배열에서 특정값을 찾아내는 알고리즘
- 미친 시간 초과 뜸 ㅋㅋ
```c
#include <algorithm>
#include <iostream>
#include <vector>
using namespace std;
int main(){
	int n;
	cin >> n;
	vector<int> a(n);
	for(int i=0; i<n; i++){
		cin >> a[i];
	}
	sort(a.begin(),a.end()); //정렬을 해야 이진탐색 가능, 안해주면 모두 0이나온다.
	int m;
	cin >> m;
	for(int i=0; i<m; i++){
		int number;
		cin >> number;
		cout<< binary_search(a.begin(), a.end(), number) << " ";
	}
	cout << '\n';
	
	return 0;
}
```
